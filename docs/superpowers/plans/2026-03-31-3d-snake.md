# 3D Snake Game Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a complete 3D Snake game in Unity from scratch using URP with glowing emission materials and Bloom post-processing.

**Architecture:** Grid-based snake game where positions are stored as `Vector2Int` and mapped to 3D world space at Y=0.5. A single `SnakeGame` MonoBehaviour owns all game state. Visual system uses URP with a custom HLSL shader for the grid floor and HDR emission materials for all game objects.

**Tech Stack:** Unity URP (`com.unity.render-pipelines.universal`), HLSL custom shader, UGUI (`UnityEngine.UI.Text`), C# MonoBehaviour

**User Verification:** YES — user visually confirms screenshot matches spec after Task 7

- [ ] **Step 2: Create GridEmission.shader**

```yaml
manage_shader:
  action: create
  name: GridEmission
  path: Assets/Shaders/
  contents: |
    Shader "Custom/GridEmission"
    {
        Properties
        {
            _GridColor ("Grid Line Color", Color) = (0.2, 0.8, 0.9, 1.0)
            _FloorColor ("Floor Color", Color) = (0.02, 0.02, 0.05, 0.9)
            _GridSize ("Grid Size (cells)", Float) = 30.0
            _LineWidth ("Line Width (fraction of cell)", Float) = 0.05
            _EmissionIntensity ("Emission Intensity", Float) = 1.5
        }
        SubShader
        {
            Tags { "RenderType"="Transparent" "Queue"="Transparent" "RenderPipeline"="UniversalPipeline" }
            Pass
            {
                Name "GridEmissionPass"
                Tags { "LightMode" = "UniversalForward" }
                Blend SrcAlpha OneMinusSrcAlpha
                ZWrite Off
                Cull Off
                HLSLPROGRAM
                #pragma vertex vert
                #pragma fragment frag
                #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"
                struct Attributes
                {
                    float4 positionOS : POSITION;
                    float2 uv : TEXCOORD0;
                };
                struct Varyings
                {
                    float4 positionHCS : SV_POSITION;
                    float2 uv : TEXCOORD0;
                };
                CBUFFER_START(UnityPerMaterial)
                    float4 _GridColor;
                    float4 _FloorColor;
                    float _GridSize;
                    float _LineWidth;
                    float _EmissionIntensity;
                CBUFFER_END
                Varyings vert(Attributes IN)
                {
                    Varyings OUT;
                    OUT.positionHCS = TransformObjectToHClip(IN.positionOS.xyz);
                    OUT.uv = IN.uv;
                    return OUT;
                }
                half4 frag(Varyings IN) : SV_Target
                {
                    // frac(uv * GridSize): each cell 0..1, 30 cells across the plane
                    float2 cell = frac(IN.uv * _GridSize);
                    // distance from nearest edge (0 = on line, 0.5 = cell center)
                    float2 dist = min(cell, 1.0 - cell);
                    // _LineWidth = fraction of cell covered by line (0.05 = 5% of 1wu = 0.05wu)
                    float onLine = step(dist.x, _LineWidth * 0.5) + step(dist.y, _LineWidth * 0.5);
                    onLine = saturate(onLine);
                    half4 lineCol = _GridColor * _EmissionIntensity;
                    lineCol.a = _GridColor.a;
                    return lerp(_FloorColor, lineCol, onLine);
                }
                ENDHLSL
            }
        }
    }
```

        {
            startY = transform.position.y;

- [ ] **Step 3: Create SnakeGame.cs**

```yaml
manage_script:
  action: create
  name: SnakeGame
  path: Assets/Scripts/SnakeGame.cs
  contents: |
    using System.Collections.Generic;
    using UnityEngine;
    using UnityEngine.UI;

    public class SnakeGame : MonoBehaviour
    {
        [Header("Prefabs")]
        public GameObject snakeHeadPrefab;
        public GameObject snakeBodyPrefab;
        public GameObject foodPrefab;

        [Header("UI")]
        public Text scoreText;

        [Header("Settings")]
        public int gridHalfSize = 10;
        public float moveInterval = 0.3f;
        public float minMoveInterval = 0.1f;
        public float speedIncrement = 0.005f;

        private Vector2Int currentDir = Vector2Int.right;
        private Vector2Int pendingDir = Vector2Int.right;

        // positions[0] = head, positions[1..n] = body
        private List<Vector2Int> positions = new List<Vector2Int>();
        // segments[i] matches positions[i]
        private List<GameObject> segments = new List<GameObject>();

        private GameObject food;
        private Vector2Int foodGridPos;
        private int score;
        private float stepTimer;
        private bool isDead;

        void Start()
        {
            // Head at origin
            positions.Add(Vector2Int.zero);
            segments.Add(Instantiate(snakeHeadPrefab, ToWorld(Vector2Int.zero), Quaternion.identity));

            // 3 initial body segments trailing left
            for (int i = 1; i <= 3; i++)
            {
                var p = new Vector2Int(-i, 0);
                positions.Add(p);
                segments.Add(Instantiate(snakeBodyPrefab, ToWorld(p), Quaternion.identity));
            }

            SpawnFood();
            UpdateScoreUI();
        }

        void Update()
        {
            if (isDead) return;
            ReadInput();
            stepTimer += Time.deltaTime;
            if (stepTimer >= moveInterval)
            {
                stepTimer = 0f;
                Step();
            }
        }

        void ReadInput()
        {
            if ((Input.GetKeyDown(KeyCode.W) || Input.GetKeyDown(KeyCode.UpArrow))
                && currentDir != Vector2Int.down)
                pendingDir = Vector2Int.up;
            else if ((Input.GetKeyDown(KeyCode.S) || Input.GetKeyDown(KeyCode.DownArrow))
                && currentDir != Vector2Int.up)
                pendingDir = Vector2Int.down;
            else if ((Input.GetKeyDown(KeyCode.A) || Input.GetKeyDown(KeyCode.LeftArrow))
                && currentDir != Vector2Int.right)
                pendingDir = Vector2Int.left;
            else if ((Input.GetKeyDown(KeyCode.D) || Input.GetKeyDown(KeyCode.RightArrow))
                && currentDir != Vector2Int.left)
                pendingDir = Vector2Int.right;
        }

        void Step()
        {
            currentDir = pendingDir;
            Vector2Int newHead = positions[0] + currentDir;

            // Wall collision
            if (Mathf.Abs(newHead.x) > gridHalfSize || Mathf.Abs(newHead.y) > gridHalfSize)
            {
                Die();
                return;
            }

            // Self collision (skip last segment — it moves away this step)
            for (int i = 0; i < positions.Count - 1; i++)
            {
                if (positions[i] == newHead)
                {
                    Die();
                    return;
                }
            }

            bool ateFood = (newHead == foodGridPos);

            if (ateFood)
            {
                // Insert new head, keep all body (tail stays = snake grows by 1)
                positions.Insert(0, newHead);
                // Add new body segment at the old tail position (now positions[Count-1])
                segments.Add(Instantiate(snakeBodyPrefab,
                    ToWorld(positions[positions.Count - 1]), Quaternion.identity));
                score += 10;
                moveInterval = Mathf.Max(minMoveInterval, moveInterval - speedIncrement);
                SpawnFood();
                UpdateScoreUI();
            }
            else
            {
                // Drop old tail, prepend new head
                positions.RemoveAt(positions.Count - 1);
                positions.Insert(0, newHead);
            }

            // Sync all GameObjects to their new positions
            for (int i = 0; i < segments.Count; i++)
                segments[i].transform.position = ToWorld(positions[i]);
        }

        void SpawnFood()
        {
            if (food != null) Destroy(food);
            Vector2Int pos;
            int tries = 0;
            do
            {
                pos = new Vector2Int(
                    Random.Range(-gridHalfSize, gridHalfSize + 1),
                    Random.Range(-gridHalfSize, gridHalfSize + 1));
                tries++;
            } while (positions.Contains(pos) && tries < 200);

            foodGridPos = pos;
            food = Instantiate(foodPrefab, ToWorld(pos), Quaternion.identity);
        }

        void Die()
        {
            isDead = true;
            if (scoreText != null)
                scoreText.text = "SCORE: " + score + "\nGAME OVER";
        }

        void UpdateScoreUI()
        {
            if (scoreText != null)
                scoreText.text = "SCORE: " + score;
        }

        // Maps 2D grid position to 3D world: Y=0.5 floats above ground plane
        Vector3 ToWorld(Vector2Int g) => new Vector3(g.x, 0.5f, g.y);
    }
```
